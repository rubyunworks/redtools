def method_missing(meth, *args)
  if shell.respond_to?(meth)
    shell.__send__(meth, *args)
  else
    super(meth, *args) if defined?(super)
  end
end

def shell(path=Dir.pwd)
  @shell ||= {}
  @shell[path] ||= (
    mode = {
      :noop    => trial?,
      :verbose => trace? || (trial? && !quiet?),
      :quiet   => quiet?
    }
    Ratch::Shell.new(path, mode)
  )
end
